%% Helpers %%

\newmacro{newrule}[4][2]
  {\newmacro{#1}{\inferrule*[lab={#1},right={$#2$}]
    {#3}
    {#4}}}
\newmacro{userule}
  {\usemacro}
\newmacro{refrule}[1]
  {\ifthenelse{\isundefined{#1}}
    {\GenericError{}{Rule `#1` is not defined}{}{}}
    {\textsc{#1}}}

\statefulfalse
\newmacro{st}[1]
  {\ifstateful{, #1}\else{}\fi}


%% Typing %%

\newmacro{RelationT}
  {\Gamma\st{\Sigma} \infers e : \tau}

\newrule{T-Fail}
  {\ }
  {\Gamma\st{\Sigma} \infers \Fail : \Task \tau}

\newrule{T-Edit}
  {\Gamma\st{\Sigma} \infers e : \beta}
  {\Gamma\st{\Sigma} \infers \Edit e : \Task \beta}

\newrule{T-Fill}
  {\ }
  {\Gamma\st{\Sigma} \infers \Fill \beta : \Task \beta}

\newrule{T-Change}
  {\Gamma, \Sigma \infers e : \beta}
  {\Gamma, \Sigma \infers \Change e : \Task \beta}

\newrule{T-Next}
  {\Gamma\st{\Sigma} \infers e_1 : \Task \tau_1 \\
   \Gamma\st{\Sigma} \infers e_2 : \tau_1 \to \Task \tau_2}
  {\Gamma\st{\Sigma} \infers e_1 \ExThen e_2 : \Task \tau_2}

\newrule{T-Then}
  {\Gamma\st{\Sigma} \infers e_1 : \Task \tau_1 \\
   \Gamma\st{\Sigma} \infers e_2 : \tau_1 \to \Task \tau_2}
  {\Gamma\st{\Sigma} \infers e_1 \Then e_2 : \Task \tau_2}

\newrule{T-All}
  {\Gamma\st{\Sigma} \infers e_1 : \Task \tau_1 \\
   \Gamma\st{\Sigma} \infers e_2 : \Task \tau_2}
  {\Gamma\st{\Sigma} \infers e_1 \All e_2 : \Task\,(\tau_1 \times \tau_2)}

\newrule{T-One}
  {\Gamma\st{\Sigma} \infers e_1 : \Task \tau \\
   \Gamma\st{\Sigma} \infers e_2 : \Task \tau }
  {\Gamma\st{\Sigma} \infers e_1 \One e_2 : \Task \tau}

%% Handling %%

\newmacro{RelationH}
  {t\st{s} \handle{\eta} t'\st{s'}}

\newrule{H-Change}[v, v' : \beta]
  { }
  {\Edit v\st{s} \handle{v'} \Edit v'\st{s}}

\newrule{H-Clear}[v : \beta]
  { }
  {\Edit v\st{s} \handle{\Clear} \Fill \beta\st{s}}

\newrule{H-Fill}[v' : \beta]
  { }
  {\Fill \beta\st{s} \handle{v'} \Edit v'\st{s}}

\newrule{H-Store}[\Sigma(l), v' : \tau]
  { }
{\Change l, s \handle{v'} \Change l, [l \mapsto v']s}

\newrule{H-Stay'}[\Observe(t_1) = \nothing]
  {\ }
  {t_1 \ExThen e\st{s} \handle{\Next} t_1 \ExThen e\st{s}}

\newrule{H-Next'}[\Observe(t_1) = v]
  {e\ v \evaluate t_2    \\
   t_2\st{s} \normalise t_2'\st{s'} }
  {t_1 \ExThen e\st{s} \handle{\Next} t_2'\st{s'}}

\newrule{H-Stay}[\Observe(t_1) = \nothing]
  {\ }
  {t_1 \Then e\st{s} \handle{\Execute \pi} t_1 \Then e\st{s}}

\newrule{H-Fail}[\Observe(t_1) = v \land t_2 = \Fail]
  {e\ v \evaluate t_2    \\
   t_2\st{s} \handle{\Pick \pi} t_2'\st{s'} }
  {t_1 \Then e\st{s} \handle{\Execute \pi} t_1 \Then e\st{s}}

\newrule{H-Next}[\Observe(t_1) = v \land t_2 \neq \Fail]
  {e\ v \evaluate t_2    \\
   t_2\st{s} \handle{\Pick \pi} t_2'\st{s'} }
  {t_1 \Then e\st{s} \handle{\Execute \pi} t_2'\st{s'}}

\newrule{H-PassS}
  {t_1\st{s} \handle{\eta} t_1'\st{s'}}
  {t_1 \ExThen e\st{s} \handle{\eta} t_1' \ExThen e\st{s'}}

\newrule{H-Pass}[\eta \neq \Execute \pi]
  {t_1\st{s} \handle{\eta} t_1'\st{s'}       \\
   t_1' \Then e\st{s'} \normalise t_2\st{s''} }
  {t_1 \Then e\st{s} \handle{\eta} t_2\st{s''}}

\newrule{H-First}[t_1 \neq \Fail]
  {\ }
  {t_1 \One t_2\st{s} \handle{\Pick \First} t_1\st{s}}

\newrule{H-Second}[t_2 \neq \Fail]
  {\ }
  {t_1 \One t_2\st{s} \handle{\Pick \Second} t_2\st{s}}

\newrule{H-Other}[t_2 \neq \Fail]
  {t_2\st{s} \handle{\Pick \pi} t_2'\st{s'}}
  {t_1 \One t_2\st{s} \handle{\Pick \Other \pi} t_2'\st{s'}}

\newrule{H-Left}
  {t_1\st{s} \handle{\eta} t_1'\st{s'} }
  {t_1 \AndOr t_2\st{s} \handle{\Left \eta} t_1' \AndOr t_2\st{s'}}

\newrule{H-Right}
  {t_2\st{s} \handle{\eta} t_2'\st{s'} }
  {t_1\st{s} \AndOr t_2 \handle{\Right \eta} t_1 \AndOr t_2'\st{s'}}

\newrule{H-Fallback}
  { }
  {t\st{s} \handle{\eta} t\st{s}}
