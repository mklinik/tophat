    {#3}
    {#4}}}
\newmacro{userule}
  {\usemacro}
\newmacro{refrule}[1]
  {\ifthenelse{\isundefined{#1}}
    {\GenericError{}{Rule `#1` is not defined}{}{}}
    {\textsc{#1}}}

%% Typing %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\newmacro{RelationT}
  {\Gamma,\Sigma \infers e : \tau}

\newrule{T-Abs}
  {\Gamma[x:\tau_1] ,\Sigma \infers e:\tau_2}
  {\Gamma,\Sigma \infers \lambda x : \tau_1 . e :\tau_1 \to \tau_2}

\newrule{T-Var}
  {x:\tau\in\Gamma}
  {\Gamma,\Sigma\infers x:\tau}

\newrule{T-If}
  {\Gamma,\Sigma \infers e_1:\Bool\\
   \Gamma,\Sigma \infers e_2:\tau\\
   \Gamma,\Sigma \infers e_3:\tau}
  {\Gamma,\Sigma \infers \If{e_1}{e_2}{e_3}:\tau}

\newrule{T-App}
  {\Gamma,\Sigma \infers e_1:\tau_1\to\tau_2\\
   \Gamma,\Sigma \infers e_2:\tau_1}
  {\Gamma,\Sigma \infers e_1 e_2 :\tau_2}


\newrule{T-Ref}
  {\Gamma,\Sigma \infers e:\tau}
  {\Gamma,\Sigma \Ref e :\Reference \tau}

\newrule{T-Deref}
  {\Gamma,\Sigma \infers e:\Reference \tau}
  {\Gamma,\Sigma\infers e:\tau}

\newrule{T-Loc}
  {\Sigma(l):\tau}
  {\Gamma,\Sigma\infers l:\tau}

\newrule{T-Assign}
  {\Gamma,\Sigma\infers e_1:\Reference \tau\\
   \Gamma,\Sigma\infers e_2:\tau}
  {\Gamma,\Sigma\infers e_1 := e_2:\Unit}

\newrule{T-Pair}
    {\Gamma,\Sigma \infers e_1 : \tau_1 \\
     \Gamma,\Sigma \infers e_2 : \tau_2}
    {\Gamma,\Sigma \infers \tuple{e_1, e_2} :\tau_1 \times \tau_2}

\newrule{T-Edit}
  {\Gamma,\Sigma \infers e : \beta}
  {\Gamma,\Sigma \infers \Edit e : \Task \beta}

\newrule{T-Fill}
  {\ }
  {\Gamma,\Sigma \infers \Enter \beta : \Task \beta}

\newrule{T-Update}
  {\Gamma,\Sigma \infers e : \Reference \beta}
  {\Gamma,\Sigma \infers \Update e : \Task \beta}


\newrule{T-Fail}
  { }
  {\Gamma,\Sigma \infers \Fail : \Task \alpha}


\newrule{T-Then}
  {\Gamma,\Sigma \infers e_1 : \Task \tau_1 \\
   \Gamma,\Sigma \infers e_2 : \tau_1 \to \Task \tau_2}
  {\Gamma,\Sigma \infers e_1 \Then e_2 : \Task \tau_2}


\newrule{T-Next}
  {\Gamma,\Sigma \infers e_1 : \Task \tau_1 \\
   \Gamma,\Sigma \infers e_2 : \tau_1 \to \Task \tau_2}
  {\Gamma,\Sigma \infers e_1 \Next e_2 : \Task \tau_2}


\newrule{T-And}
  {\Gamma,\Sigma \infers e_1 : \Task \tau_1 \\
   \Gamma,\Sigma \infers e_2 : \Task \tau_2}
  {\Gamma,\Sigma \infers e_1 \And e_2 : \Task\,(\tau_1 \times \tau_2)}


\newrule{T-Or}
  {\Gamma,\Sigma \infers e_1 : \Task \tau \\
   \Gamma,\Sigma \infers e_2 : \Task \tau }
  {\Gamma,\Sigma \infers e_1 \Or e_2 : \Task \tau}


\newrule{T-Xor}
  {\Gamma,\Sigma \infers e_1 : \Task \tau \\
   \Gamma,\Sigma \infers e_2 : \Task \tau }
  {\Gamma,\Sigma \infers e_1 \Xor e_2 : \Task \tau}



%% Evaluation %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\newmacro{RelationV}
  {e \evaluate v}


\newrule{E-Edit}
  {e \evaluate v}
  {\Edit e \evaluate \Edit v}

\newrule{E-Fill}
  {\ }
  {\Enter \beta \evaluate \Enter \beta}

\newrule{E-Update}
  {e \evaluate l}
  {\Update e \evaluate \Update l}


\newrule{E-Fail}
  { }
  {\Fail \evaluate \Fail}


\newrule{E-Then}
  {e_1 \evaluate t_1}
  {e_1 \Then e_2 \evaluate t_1 \Then e_2}

\newrule{E-Next}
  {e_1 \evaluate t_1}
  {e_1 \Next e_2 \evaluate t_1 \Next e_2}


\newrule{E-And}
  {e_1 \evaluate t_1 \\
   e_2 \evaluate t_2}
  {e_1 \And e_2 \evaluate t_1 \And t_2}


\newrule{E-Or}
  {e_1 \evaluate t_1 \\
   e_2 \evaluate t_2}
  {e_1 \Or e_2 \evaluate t_1 \Or t_2}

\newrule{E-Xor}
  {\ }
  {e_1 \Xor e_2 \evaluate e_1 \Xor e_2}





%% Normalisation %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\newmacro{RelationN}
  {t,\Sigma \normalise t'\st{'}}


\newrule{N-Edit}
  { }
  {\Edit v,\Sigma \normalise \Edit v,\Sigma}

\newrule{N-Fill}
  { }
  {\Enter \beta,\Sigma \normalise \Enter \beta,\Sigma}

\newrule{N-Update}
  { }
  {\Update l,\Sigma \normalise \Update l,\Sigma}


\newrule{N-Fail}
  { }
  {\Fail,\Sigma \normalise \Fail,\Sigma}


\newrule{N-ThenStay}[\Value(t_1') = \bot]
  {t_1,\Sigma \normalise t_1'\st{'}}
  {t_1 \Then e_2,\Sigma \normalise t_1' \Then e_2\st{'}}

\newrule{N-ThenFail}[\Value(t_1') = v_1 \land \lnot\Failing(t_2)]
  {t_1,\Sigma \normalise t_1'\st{'} \\
   e_2\ v_1 \evaluate t_2}
  {t_1 \Then e_2,\Sigma \normalise t_1' \Then e_2\st{'}}

\newrule{N-ThenCont}[\Value(t_1') = v_1 \land \Failing(t_2)]
  {t_1,\Sigma \normalise t_1'\st{'} \\
   e_2\ v_1 \evaluate t_2  \\
   t_2\st{'} \normalise t_2'\st{''}}
  {t_1 \Then e_2,\Sigma \normalise t_2'\st{''}}

\newrule{N-Next}
  {t_1,\Sigma \normalise t_1'\st{'}}
  {t_1 \Next e_2,\Sigma \normalise t_1' \Next e_2\st{'}}


\newrule{N-And}
  {t_1,\Sigma  \normalise t_1'\st{'} \\
   t_2\st{'} \normalise t_2'\st{''}}
  {t_1 \And t_2,\Sigma \normalise t_1' \And t_2'\st{''}}


\newrule{N-OrLeft}[\Value(t_1') = v_1]
  {t_1,\Sigma  \normalise t_1'\st{'}}
  {t_1 \Or t_2,\Sigma \normalise t_1'\st{'}}

\newrule{N-OrRight}[\Value(t_1') = \bot \land \Value(t_2') = v_2]
  {t_1,\Sigma  \normalise t_1'\st{'} \\
   t_2\st{'} \normalise t_2'\st{''}}
  {t_1 \Or t_2,\Sigma \normalise t_2'\st{''}}

\newrule{N-OrNone}[\Value(t_1') = \bot \land \Value(t_2') = \bot]
  {t_1,\Sigma  \normalise t_1'\st{'} \\
   t_2\st{'} \normalise t_2'\st{''}}
  {t_1 \Or t_2,\Sigma \normalise t_1' \Or t_2'\st{''}}


\newrule{N-Xor}
  { }
  {e_1 \Xor e_2,\Sigma \normalise e_1 \Xor e_2,\Sigma}

\newrule{N-Eval}
    {e\evaluate e' \\
    e\neq e'}
    {e \normalise e'}

%% Handling %%


\newmacro{RelationH}
  {t,\Sigma \handle{i} t'\st{'}}


\newrule{H-Change}[v, v' : \beta]
  { }
  {\Edit v,\Sigma \handle{v'} \Edit v',\Sigma}

\newrule{H-Empty}[v : \beta]
  { }
  {\Edit v,\Sigma \handle{\Empty} \Enter \beta,\Sigma}

\newrule{H-Fill}[v' : \beta]
  { }
  {\Enter \beta,\Sigma \handle{v'} \Edit v',\Sigma}

\newrule{H-Update}[\Sigma(l), v' : \beta]
  { }
  {\Update l,\Sigma \handle{v'} \Update l\st[l \mapsto v']{}}


\newrule{H-Fail}
  { }
  {\Fail,\Sigma \handle{i} \Fail,\Sigma}


\newrule{H-PassThen}
  {t_1,\Sigma \handle{i} t_1'\st{'}}
  {t_1 \Then e_2,\Sigma \handle{i} t_1' \Then e_2\st{'}}

\newrule{H-PassNext}
  {t_1,\Sigma \handle{i} t_1'\st{'}}
  {t_1 \Next e_2,\Sigma \handle{i \neq \Continue} t_1' \Next e_2\st{'}}

\newrule{H-Next}[\Value{(t_1)}\neq \bot \wedge \neg\Failing{(e_2 \Value{(t_1)}\normalise)}]
  { }
  {t_1 \Next e_2,\Sigma \handle{\Continue} t_1 \Then e_2,\Sigma}


\newrule{H-FirstAnd}
  {t_1,\Sigma \handle{i} t_1'\st{'} }
  {t_1 \And t_2,\Sigma \handle{\First i} t_1' \And t_2\st{'}}

\newrule{H-SecondAnd}
  {t_2,\Sigma \handle{i} t_2'\st{'} }
  {t_1,\Sigma \And t_2 \handle{\Second i} t_1 \And t_2'\st{'}}


\newrule{H-FirstOr}
  {t_1,\Sigma \handle{i} t_1'\st{'} }
  {t_1 \Or t_2,\Sigma \handle{\First i} t_1' \Or t_2\st{'}}

\newrule{H-SecondOr}
  {t_2,\Sigma \handle{i} t_2'\st{'} }
  {t_1 \Or t_2,\Sigma \handle{\Second i} t_1 \Or t_2'\st{'}}


\newrule{H-PickLeft}[\Failing(t_1)]
  {t_1,\Sigma \handle{\Pick r} t_1'\st{'}}
  {t_1 \Xor t_2,\Sigma \handle{\Pick \Left r} t_1'\st{'}}

\newrule{H-PickRight}[\Failing(t_2)]
  {t_2,\Sigma \handle{\Pick r} t_2'\st{'}}
  {t_1 \Xor t_2,\Sigma \handle{\Pick \Right r} t_2'\st{'}}

\newrule{H-PickHere}
  {\ }
  {t,\Sigma \handle{\Pick \Here} t,\Sigma}
