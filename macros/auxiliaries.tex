

%% Definitions %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%% \(re)newmacro{.name.}[.num.][.list.]{.def.}
\newcommand*{\newmacro}[1]
  {\expandafter\newcommandx\expandafter*\csname #1\endcsname}
\newcommand*{\renewmacro}[1]
  {\expandafter\renewcommandx\expandafter*\csname #1\endcsname}

%% \usemacro{.name.}
\newcommand*{\usemacro}[1]
  {\csname #1\endcsname}


%% Tests %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newmacro{isempty}[1]
  {\equal{#1}{}}


%% Math %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%% \(re)newmathcommand[.cat.]{.name.}{.def.}
\newmacro{newmathcommand}[3][1]
  {\ifthenelse{\isempty{#1}}
    {\newmacro{#2}{#3}}
    {\newmacro{#2}{\usemacro{math#1}{#3}}}}
\newmacro{renewmathcommand}[3][1]
  {\ifthenelse{\isempty{#1}}
    {\renewmacro{#2}{#3}}
    {\renewmacro{#2}{\usemacro{math#1}{#3}}}}
