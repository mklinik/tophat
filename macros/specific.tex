%% Host language %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newkeyword[IF]  {if}
\newkeyword[THEN]{then}
\newkeyword[ELSE]{else}

\newkeyword[Ref] {ref}

\newmacro{If}[3]
  {\IF #1 \THEN #2 \ELSE #3}


%% Values %%

\newvalue{True}
\newvalue{False}

\newmacro{astring}[1]
  {\text{``#1''}}


%% Types %%

\newtype{Unit}
\newtype{Bool}
\newtype{Int}
\newtype{String}
\newtype[Reference]{Ref}

\newoperator{Next} {\vartriangleright}

%% Events %%

\newvalue[Continue]{C}

\newvalue[First]  {F}

\newvalue[Other]  {S}

  {V}
