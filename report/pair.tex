\subsection{Intermezzo: monoidal functor and its laws}

\begin{equation*}
  \begin{array}{rclr}
    \Map \Snd\ (\Edit \{\} \And t)
      &\approxeq& t
      & \text{(left identity)} \\
    \Map \Fst\ (t \And \Edit \{\})
      &\approxeq& t
      & \text{(right identity)} \\
    \Map \Assoc\ (r \And\ (s \And t))
      &\approxeq& (r \And s) \And t
      & \text{(associativity)} \\
    \Map\ (\lambda \{x, y\}. \{g\ x, h\ y\})\ (s \And t)
      &\approxeq& \Map\ g\ s \And \Map\ h\ t
      & \text{(naturality)} \\
  \end{array}
\end{equation*}
