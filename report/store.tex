    & t    & ::=& \ldots          & \\
    &      &\mid& \Store l        & – store task \\
  Types
    & \tau & ::=& \ldots          & \\
    &      &\mid& \Unit           & – unit \\
    &      &\mid& \Reference \tau & – reference type \\
\end{grammar}
