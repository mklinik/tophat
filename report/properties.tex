\subsection{Equality}

\begin{align*}
\forall t_1, t_2 .  t_1 \sim t_2 \iff \forall i . &        & t_1 \handle{i} t_1'\\
                                                  & \wedge & t_2 \handle{i} t_2'\\

\end{align*}

\begin{align*}

\end{align*}
